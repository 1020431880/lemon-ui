2. 主题样式

主题效果只包含alert，confirm，prompt，msg，需要引用对应的样式，注：iframe弹出窗口只有默认的样式

3.1 代码

// 插件参数
Dialogbox.DEFAULTS = {
    title: "标题", // 标题
    icon: "", // 成功：success，失败：error，警告：warning，提示：info，帮助：help
    content: "", // 内容
    placeholder: "", // prompt输入框提示文字
    btn: ["确定", "取消"], // 定义按钮的名称
    closebtn: true, // 是否显示右上角关闭按钮，true和false
    skin: "dialogbox-default", // 使用的皮肤样式
    onClick: "", // 点击确定按钮的回调函数
    onCancel: "", // 点击取消按钮的回调函数
    backdrop: false, // 第一个参数控制遮罩层显示，第二个控制遮罩层背景透明度，第三个是给遮罩层添加动画
    animate: [true, "animated zoomIn"], // 显示时候是否有动画效果，可以直接使用ui中定义的动画
    timeout: 1500, // 定时关闭时间
    zIndex: 100 // 设置z-index，解决因别的层的index比插件的高而不显示插件的问题

};

// 插件调用
$("#btn-dialogbox-confirm").click(function() {
    $.dialogbox.confirm({
        title: "确认框",
        icon: "info",
        content: "这是一个confirm确认框！",
        onClick: function() {
            $.dialogbox.alert({
                title: "操作提示",
                content: "您的订单已经提交成功！",
                icon: "success"
            });
        },
        onCancel: function() {
            $.dialogbox.alert({
                title: "操作提示",
                content: "您的订单已经取消！",
                icon: "info"
            });
        }
    });
});
