1.1 代码

	Datepicker.DEFAULTS = {
		skin: "datepicker-default",
		viewModel: "date", // 默认显示的视图，有year,month,yearMonth,time
		language: "zh", // 显示语言，zh：中文，en：英文
		date: "", // 默认时间，默认是当天
		minDate: "1900-01-01 00:00:00", // 最小日期
		maxDate: "2099-12-31 23:59:59", // 最大日期
		format: "yyyy-MM-dd", // 日期格式化
		festival: false, // 是否显示节日
		onBefore: function() {}, // 日期显示之前函数
		onSelect: function(date) {}, // 选择日期函数
		onAfter: function() {}, // 选择之后的函数
		fixed: false, // 是否固定显示
		marks: [], //给日期添加标注，格式是：2017-05-23
		isTime: false, // 是否显示时分秒
		zIndex: 1000 // z-index值

	};
